def method_missing(method_name, *args, &block)
  if method_name.to_s =~ /^is_(.*)\?$/ && PZA::Role.underscored_role_names.include?($1)
    self.class.send :define_method, method_name do
      self.role?($1)
    end
    self.send method_name
  else
    super
  end
end
